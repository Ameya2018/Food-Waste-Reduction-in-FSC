                production = np.random.randint(500000, 2000000)
                arrivals = np.random.randint(100, 5000)
                price = np.random.randint(3000, 8000)

            if state in ["Punjab", "Haryana", "Uttar Pradesh"]:
                if date.month in [12, 1, 2]:
                    temperature = np.random.randint(5, 15)
                    rainfall = np.random.randint(0, 30)
                elif date.month in [6, 7, 8]:
                    temperature = np.random.randint(25, 35)
                    rainfall = np.random.randint(150, 300)
                else:
                    temperature = np.random.randint(25, 40)
                    rainfall = np.random.randint(0, 50)
            elif state in ["Tamil Nadu", "Kerala"]:
                if date.month in [12, 1, 2]:
                    temperature = np.random.randint(20, 25)
                    rainfall = np.random.randint(50, 100)
                elif date.month in [6, 7, 8]:
                    temperature = np.random.randint(25, 35)
                    rainfall = np.random.randint(200, 400)

                    "Date": date,

                }
            )
        batch_df = pd.DataFrame(batch_data)
        dfs.append(batch_df)

max_data = {crops[c]: {} for c in crops}
for c in crops:
    for s in season:
        c_df = df[(df["Crop"] == crops[c]) & (df["Season"] == season[s])]
        max_data[crops[c]][season[s]] = [c_df["Price"].max(), c_df["Arrivals"].max()]
new_df = []
for v in df.values:
    pf = 1 - (v[4] / max_data[v[1]][v[3]][0])
    af = 1 - (v[5] / max_data[v[1]][v[3]][1])
    new_df.append([*v, v[6] * pf * af])
df = pd.DataFrame(new_df, columns=[*list(df.columns), "Demand"])
df.to_csv('Data/demand_data.csv', index=False)

    ax.set_xlabel("Date")
    ax.set_ylabel("Values")
    ax.legend()
