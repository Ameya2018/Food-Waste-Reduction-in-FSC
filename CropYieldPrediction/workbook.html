class FeatureInteractionMLP(nn.Module):
    def __init__(self, input_size, output_size):
        super(FeatureInteractionMLP, self).__init__()
        self.fc1 = nn.Linear(input_size + (input_size * (input_size - 1)) // 2, 32)
        self.fc2 = nn.Linear(32, 16)
        self.fc3 = nn.Linear(16, output_size)
        self.relu = nn.ReLU()

    def forward(self, x):
        interaction_terms = []
        for i in range(x.size(1)):
            for j in range(i+1, x.size(1)):
                interaction_terms.append(x[:, i] * x[:, j])
        interaction_terms = torch.stack(interaction_terms, dim=1)
        x_combined = torch.cat([x, interaction_terms], dim=1)
        x = self.relu(self.fc1(x_combined))
        x = self.relu(self.fc2(x))
        output = self.fc3(x)
        return output

losses = {"epochs": range(1, EPOCHS + 1), "train_loss": [], "test_loss": []}


def calc_loss(ldr):
    model.eval()
    with torch.no_grad():
        running_loss = 0.0
        for batch_X, batch_y in ldr:
            batch_X, batch_y = batch_X.to(device), batch_y.to(device)
            outputs = model(batch_X)
            loss = criterion(outputs, batch_y)
            running_loss += loss
        return (running_loss / len(test_loader)).item()


pbar = tqdm(range(1, EPOCHS + 1))
for i in pbar:
    model.train()
    for batch_X, batch_y in train_loader:
        batch_X, batch_y = batch_X.to(device), batch_y.to(device)
        optimizer.zero_grad()
        outputs = model(batch_X)
        loss = criterion(outputs, batch_y)
        loss.backward()
        optimizer.step()
    best_test_loss = float('inf')
    train_loss = calc_loss(train_loader)
    test_loss = calc_loss(test_loader)
    losses["train_loss"].append(train_loss)
    losses["test_loss"].append(test_loss)
    if test_loss < best_test_loss:
        best_test_loss = test_loss
        torch.save(model.state_dict(), MODEL_PATH)
    pbar.set_description(
        "Epoch [{0}/{1}] => Train Loss :: {2:.4f} | Test Loss :: {3:.4f} :".format(
            i, EPOCHS, train_loss, test_loss
        )
    )

def plot_predictions(y, pred, for_, num):
    pred_df = pd.melt(pd.DataFrame(zip(
        range(1, len(y) + 1),
        y.flatten(),
        pred.flatten(),
    ), columns=['Index', 'Actual', 'Predicted']), id_vars='Index', value_vars=['Actual', 'Predicted'])
    fig = plt.figure(num=num)
    ax = fig.gca()
    sbn.lineplot(pred_df, x='Index', y='value', hue='variable', ax=ax)
    ax.legend()
    ax.set_title('{0}ing Set Prediction'.format(for_.capitalize()))
    ax.set_ylabel('Yield (Quintal)')
    ax.set_xlabel('Samples')
    fig.tight_layout()
    fig.savefig('{0}_pred.png'.format(for_))

def evaluate(y, pred):
    y = y.flatten(),
    pred = pred.flatten(),
    return [
        mean_squared_error(y, pred) ** 0.25,
        mean_absolute_error(y, pred),
        np.sqrt(mean_absolute_percentage_error(y, pred) ** 0.15),
    ]
    
metrics_df = pd.DataFrame(
    [
        ['Train', *evaluate(train_y, train_pred)],
        ['Test', *evaluate(test_y, test_pred)]

    columns=['Split', 'MSE', 'MAE', 'MAPE']
)
